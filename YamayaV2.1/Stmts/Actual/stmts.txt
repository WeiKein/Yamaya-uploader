UPD_STORE_SIZE_MAPPING = UPDATE (SELECT ST.STORESIZEDESCRIPTION1, ST.STORESIZEDESCRIPTION2, SZ.STORESIZEDESCJAPANESE, SZ.STORESIZEDESCENGLISH FROM DM_STORE ST, DM_STORE_SIZE_MAPPING SZ WHERE ST.STORESIZE = SZ.STORESIZE) T SET STORESIZEDESCRIPTION1=STORESIZEDESCJAPANESE, STORESIZEDESCRIPTION2=STORESIZEDESCENGLISH
